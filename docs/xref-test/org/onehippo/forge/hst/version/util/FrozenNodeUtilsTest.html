16  package org.onehippo.forge.hst.version.util;
17  
18  import static org.junit.Assert.assertEquals;
19  import static org.junit.Assert.assertFalse;
20  import static org.junit.Assert.assertSame;
21  import static org.junit.Assert.assertTrue;
22  
23  import java.util.Calendar;
24  
25  import javax.jcr.Node;
26  
27  import org.junit.Before;
28  import org.junit.Test;
29  import org.onehippo.repository.mock.MockNode;
30  
31  public class FrozenNodeUtilsTest {
32  
33      private MockNode frozenNode;
34   
35      @Before
36      public void before() throws Exception {
37          MockNode versionStorage = MockNode.root().addNode("versionStorage1", "rep:versionStorage");
38          MockNode versionHistory = versionStorage.addNode("versionHistory1", "nt:versionHistory");
39          MockNode version = versionHistory.addNode("1.0", "nt:version");
40  
41          frozenNode = version.addNode("jcr:frozenNode", "nt:frozenNode");
42          frozenNode.setProperty("jcr:frozenPrimaryType", "ns1:service");
43          frozenNode.setProperty("ns1:title", "MRI");
44          frozenNode.setProperty("ns1:variableid", "mri");
45  
46          assertTrue(frozenNode.isNodeType("nt:frozenNode"));
47          assertFalse(frozenNode.isNodeType("ns1:service"));
48          assertEquals("nt:frozenNode", frozenNode.getPrimaryNodeType().getName());
49  
50          MockNode description = frozenNode.addNode("ns1:description", "nt:frozenNode");
51          description.setProperty("jcr:frozenPrimaryType", "hippostd:html");
52          description.setProperty("hippostd:content", "<p>Magnetic Resonance imaging</p>");
53  
54          assertTrue(description.isNodeType("nt:frozenNode"));
55          assertFalse(description.isNodeType("hippostd:html"));
56          assertEquals("nt:frozenNode", description.getPrimaryNodeType().getName());
57      }
58  
59      @Test
60      public void testPretendNotFrozen() throws Exception {
61          Calendar now = Calendar.getInstance();
62          NonFrozenPretenderNode pretenderNode = FrozenNodeUtils.getNonFrozenPretenderNode(frozenNode, now, now);
63  
64          assertSame(frozenNode, pretenderNode.getFrozenNode());
65          assertEquals(now, pretenderNode.getCreated());
66          assertEquals(now, pretenderNode.getAsOf());
67  
68          assertTrue(pretenderNode.isNodeType("nt:frozenNode"));
69          assertTrue(pretenderNode.isNodeType("ns1:service"));
70          assertEquals("ns1:service", pretenderNode.getPrimaryNodeType().getName());
71  
72          assertEquals("MRI", pretenderNode.getProperty("ns1:title").getString());
73          assertEquals("mri", pretenderNode.getProperty("ns1:variableid").getString());
74  
75          assertTrue(pretenderNode.hasNode("ns1:description"));
76          Node descriptionNode = pretenderNode.getNode("ns1:description");
77          assertTrue(descriptionNode instanceof NonFrozenPretenderNode);
78  
79          NonFrozenPretenderNode pretenderDescriptionNode = (NonFrozenPretenderNode) descriptionNode;
80          assertSame(frozenNode.getNode("ns1:description"), pretenderDescriptionNode.getFrozenNode());
81          assertTrue(pretenderDescriptionNode.isNodeType("nt:frozenNode"));
82          assertTrue(pretenderDescriptionNode.isNodeType("hippostd:html"));
83          assertEquals("<p>Magnetic Resonance imaging</p>",
84                       pretenderDescriptionNode.getProperty("hippostd:content").getString());
85      }

87  }
